
DROP TABLE `user`;
DROP TABLE `rooms`;

CREATE TABLE user (
	id INTEGER,
	First_Name CHAR(12),
	Last_Name CHAR(12),
	Password VARCHAR(12),
	Email VARCHAR(20),
	PRIMARY KEY(id)
);

CREATE TABLE rooms (
	rid INTEGER,
	Score INTEGER,
	PRIMARY KEY(rid)
);


CREATE TABLE comment (
	id INTEGER,
	rid INTEGER,
	Time TIMESTAMP,
	Date DATE,
	Text TEXT(250),
	PRIMARY KEY(id, rid),
	FOREIGN KEY(id) REFERENCES user,
	FOREIGN KEY(rid) REFERENCES rooms
);
